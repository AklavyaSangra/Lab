Write a c program to check whether a number is even is odd.


    #include<stdio.h>
    void main()
    {
    int a;
    printf("Enter the number:");
    scanf("%d",&a);
    if(a%2==0)
    {
        printf("The number is even",&a);
    }
    else
    {
        printf("The number is odd",&a);
    }

    }
